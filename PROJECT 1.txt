    DocumentRoot /var/www/hola
    ServerName hola.local
    ErrorLog ${APACHE_LOG_DIR}/error.log
    CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>

# Habilitar el sitio y recargar Apache

# üß± Paso 1: Instalar Postfix y dependencias

# Durante la instalaci√≥n:

# - Tipo de configuraci√≥n: Internet Site
# - Nombre del sistema de correo: puedes usar mail.local o josuel.local

# üßæ Paso 2: Editar configuraci√≥n principal de Postfix

# Agrega al final del archivo:

smtp_sasl_password_maps = hash:/etc/postfix/sasl_passwd

sudo chown root:root /etc/postfix/sasl_passwd /etc/postfix/sasl_passwd.db
sudo chmod 600 /etc/postfix/sasl_passwd /etc/postfix/sasl_passwd.db

# üîÑ Paso 5: Reiniciar Postfix
